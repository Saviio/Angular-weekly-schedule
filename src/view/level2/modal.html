Modal

An element displays content that temporarily blocks interactions with the main view of a site.

Usage

... ... ... let modalIns = modal.confirm({ content:'Please click ok to show another modal.', width:300 }) modalIns.prevent = true modalIns.ok.then((ok, dismiss) => { ok.loading() return new Promise(resolve => { setTimeout(() => { modal.normal({ content: 'Sorry to have kept you waiting.', width:240 }) resolve() }, 3000) }).then(() => { ok.done() modalIns.close() }) }) modalIns.dismiss.then(() => modalIns.close()) ... let option = { template: '#modalTemplate', controller:'demoModal', controllerAs:'Demo' } modal.open(option)
